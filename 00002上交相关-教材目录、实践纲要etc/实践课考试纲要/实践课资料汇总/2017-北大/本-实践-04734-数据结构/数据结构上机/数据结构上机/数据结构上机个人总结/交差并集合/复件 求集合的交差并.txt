#include<stdlib.h>
#include<stdio.h>
#include<iostream.h>
#include<string.h>

//冒泡排序从小到大
void BSort(int a[],int n){
	int i,j,temp;
	int ex;
	
	for(i=0;i<n-1;i++){
		ex = 0;
		for(j=n-2;j>=i;j--){
			if(a[j+1]<a[j]){
				temp = a[j+1];
				a[j+1] = a[j];
				a[j] = temp;
				ex = 1;
			}
		}
		if(ex==0){
			break;
		}
	}
}

//去掉数组中重复元素，然后返回剩余元素个数,3个for循环
int DisSame(int a[],int n){
	int w,x,y;
	for(w=0;w<n;w++){
		for(x=w+1;x<n;x++){
			if(a[x] == a[w]){
				n--;
				for(y=x;y<n;y++){
					a[y] = a[y+1];
				}
				x--;
			}
		}
	}
	return n;
}

//求交集
int JJ(int a[],int b[],int c[],int na,int nb){
	int i,j,k=0;
	for(i=0;i<na;i++){
		for(j=0;j<nb;j++){
			if(b[j] == a[i]){
				c[k++] = a[i];
				break;
			}
		}
	}
	return k;
}

//求差集，a 对 b
int CJ(int a[],int b[],int c[],int na,int nb){
	int i,j,k=0;
	int fg;
	for(i=0;i<na;i++){
		fg = 0;
		for(j=0;j<nb;j++){
			if(b[j] == a[i]){
				fg = 1;
				break;
			}
		}
		if(fg==0){
			c[k++] = a[i];
		}
	}
	return k;
}

void print(int a[],int na){
	for(int i=0;i<na;i++){
		cout<< a[i] <<" ";
	}
	cout<<endl;
}

void main(){
	int a[100],b[100],c[200];
	int na=0,nb=0,nc=0,i;

	cout<<"请输入数组a,之间用空格隔开，以-9999结束"<<endl;
	for(i=0;i<100;i++){
		scanf("%d",&a[i]);
		if(a[i]==-9999){
			break;
		}
	}
	na = i;
	BSort(a,na);
	na = DisSame(a,na);
	print(a,na);

	cout<<"请输入数组b,之间用空格隔开，以-9999结束"<<endl;
	for(i=0;i<100;i++){
		scanf("%d",&b[i]);
		if(b[i]==-9999){
			break;
		}
	}
	nb = i;
	BSort(b,nb);
	nb = DisSame(b,nb);
	print(b,nb);

	nc=JJ(a,b,c,na,nb);
	cout<<"交集："<<endl;
	print(c,nc);

	
	nc=CJ(a,b,c,na,nb);
	cout<<"差集："<<endl;
	print(c,nc);
	getchar();
}
